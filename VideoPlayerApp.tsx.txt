import React, { useState, useRef, useEffect } from 'react';
import { Play, Pause, Volume2, VolumeX, Maximize, RotateCcw, Download, SkipBack, SkipForward, Settings, Zap, Lock, Shield } from 'lucide-react';
import { SecurityService } from './SecurityService';

export default function VideoPlayerApp() {
  const videoRef = useRef(null);
  const [isPlaying, setIsPlaying] = useState(false);
  const [videoUrl, setVideoUrl] = useState('');
  const [duration, setDuration] = useState(0);
  const [currentTime, setCurrentTime] = useState(0);
  const [isMuted, setIsMuted] = useState(false);
  const [volume, setVolume] = useState(1);
  const [showUrlInput, setShowUrlInput] = useState(true);
  const [isFullscreen, setIsFullscreen] = useState(false);
  const [isDownloading, setIsDownloading] = useState(false);
  const [downloadProgress, setDownloadProgress] = useState(0);
  const [showSettings, setShowSettings] = useState(false);
  const [brightness, setBrightness] = useState(100);
  const [contrast, setContrast] = useState(100);
  const [saturation, setSaturation] = useState(100);
  const [bass, setBass] = useState(1);
  const [treble, setTreble] = useState(1);
  const [showSecurityInfo, setShowSecurityInfo] = useState(false);
  const containerRef = useRef(null);

  // Uygulama başladığında ayarları yükle
  useEffect(() => {
    loadSettings();
  }, []);

  // Ayarları yerel depolamadan yükle (Şifrelenmiş)
  const loadSettings = async () => {
    try {
      const savedSettings = await SecurityService.getPreferences('videoSettings');
      if (savedSettings) {
        setBrightness(savedSettings.brightness || 100);
        setContrast(savedSettings.contrast || 100);
        setSaturation(savedSettings.saturation || 100);
        setBass(savedSettings.bass || 1);
        setTreble(savedSettings.treble || 1);
      }
    } catch (error) {
      console.error('Ayarları yükleme hatası:', error);
    }
  };

  // Ayarları kaydet (Şifrelenmiş)
  const saveSettings = async () => {
    try {
      await SecurityService.savePreferences('videoSettings', {
        brightness,
        contrast,
        saturation,
        bass,
        treble,
        lastUpdated: new Date().toISOString()
      });
    } catch (error) {
      console.error('Ayarları kaydetme hatası:', error);
    }
  };

  // Parlaklık değiştiğinde kaydet
  useEffect(() => {
    saveSettings();
  }, [brightness, contrast, saturation, bass, treble]);

  const togglePlayPause = () => {
    if (videoRef.current) {
      if (isPlaying) {
        videoRef.current.pause();
      } else {
        videoRef.current.play();
      }
      setIsPlaying(!isPlaying);
    }
  };

  const skipForward = () => {
    if (videoRef.current) {
      videoRef.current.currentTime = Math.min(videoRef.current.currentTime + 10, duration);
    }
  };

  const skipBackward = () => {
    if (videoRef.current) {
      videoRef.current.currentTime = Math.max(videoRef.current.currentTime - 10, 0);
    }
  };

  const handleTimeUpdate = () => {
    if (videoRef.current) {
      setCurrentTime(videoRef.current.currentTime);
    }
  };

  const handleLoadedMetadata = () => {
    if (videoRef.current) {
      setDuration(videoRef.current.duration);
    }
  };

  const handleProgressChange = (e) => {
    const time = parseFloat(e.target.value);
    setCurrentTime(time);
    if (videoRef.current) {
      videoRef.current.currentTime = time;
    }
  };

  const toggleMute = () => {
    if (videoRef.current) {
      videoRef.current.muted = !isMuted;
      setIsMuted(!isMuted);
    }
  };

  const handleVolumeChange = (e) => {
    const vol = parseFloat(e.target.value);
    setVolume(vol);
    if (videoRef.current) {
      videoRef.current.volume = vol;
      if (vol > 0 && isMuted) {
        setIsMuted(false);
        videoRef.current.muted = false;
      }
    }
  };

  const toggleFullscreen = () => {
    if (!document.fullscreenElement) {
      containerRef.current?.requestFullscreen().catch((err) => {
        console.error(`Tam ekran hatası: ${err.message}`);
      });
      setIsFullscreen(true);
    } else {
      document.exitFullscreen();
      setIsFullscreen(false);
    }
  };

  const resetVideo = () => {
    setShowUrlInput(true);
    setVideoUrl('');
    setIsPlaying(false);
    setCurrentTime(0);
    setDuration(0);
    setDownloadProgress(0);
    setBrightness(100);
    setContrast(100);
    setSaturation(100);
    setBass(1);
    setTreble(1);
    setShowSettings(false);
    if (videoRef.current) {
      videoRef.current.src = '';
    }
  };

  const handleVideoUrlSubmit = () => {
    if (videoUrl.trim()) {
      setShowUrlInput(false);
      setIsPlaying(false);
      setCurrentTime(0);
    }
  };

  const downloadVideo = async () => {
    if (!videoUrl.trim()) {
      alert('Lütfen geçerli bir video URL\'si girin');
      return;
    }

    setIsDownloading(true);
    setDownloadProgress(0);

    try {
      const response = await fetch(videoUrl, {
        mode: 'cors',
        credentials: 'omit'
      });

      if (!response.ok) {
        throw new Error(`HTTP error! status: ${response.status}`);
      }

      const contentLength = response.headers.get('content-length');
      const total = parseInt(contentLength, 10);

      const reader = response.body.getReader();
      const chunks = [];
      let receivedLength = 0;

      while (true) {
        const { done, value } = await reader.read();
        if (done) break;

        chunks.push(value);
        receivedLength += value.length;

        if (total) {
          const progress = Math.round((receivedLength / total) * 100);
          setDownloadProgress(progress);
        }
      }

      const blob = new Blob(chunks, { type: 'video/mp4' });
      const url = window.URL.createObjectURL(blob);
      const link = document.createElement('a');
      link.href = url;
      
      const urlParts = videoUrl.split('/');
      const fileName = urlParts[urlParts.length - 1].split('?')[0] || 'video.mp4';
      
      link.download = fileName;
      document.body.appendChild(link);
      link.click();
      document.body.removeChild(link);
      window.URL.revokeObjectURL(url);

      setDownloadProgress(0);
      setIsDownloading(false);
      alert('✓ Video başarıyla indirildi!');
    } catch (error) {
      console.error('İndirme hatası:', error);
      setIsDownloading(false);
      setDownloadProgress(0);
      alert(`❌ İndirme hatası: ${error.message}`);
    }
  };

  const formatTime = (time) => {
    if (!time) return '0:00';
    const minutes = Math.floor(time / 60);
    const seconds = Math.floor(time % 60);
    return `${minutes}:${seconds.toString().padStart(2, '0')}`;
  };

  return (
    <div 
      ref={containerRef}
      className={`flex flex-col bg-black text-white ${isFullscreen ? 'w-screen h-screen' : 'w-full max-w-full'}`}
      style={{ fontFamily: 'system-ui, -apple-system, sans-serif' }}
    >
      {/* Güvenlik Bilgisi Banner */}
      {!showUrlInput && (
        <div className="bg-green-900 bg-opacity-50 px-3 py-2 flex items-center gap-2 text-xs text-green-200">
          <Shield size={16} />
          <span>🔒 Tüm veriler şifrelenmiş ve cihazda depolanmaktadır</span>
        </div>
      )}

      {/* URL Input Section */}
      {showUrlInput && (
        <div className="flex flex-col items-center justify-center flex-1 p-4 bg-gradient-to-br from-slate-900 to-black min-h-screen">
          <div className="w-full max-w-md space-y-4">
            <div className="flex items-center justify-center mb-8">
              <Lock size={32} className="text-green-500 mr-3" />
              <h1 className="text-3xl font-bold">Video Oynatıcı Pro</h1>
            </div>
            
            <div className="space-y-2">
              <label className="block text-sm font-medium">Video URL'si girin:</label>
              <input
                type="text"
                value={videoUrl}
                onChange={(e) => setVideoUrl(e.target.value)}
                placeholder="https://example.com/video.mp4"
                className="w-full px-4 py-3 rounded-lg bg-slate-800 text-white placeholder-slate-400 border border-slate-700 focus:outline-none focus:border-blue-500"
                onKeyPress={(e) => e.key === 'Enter' && handleVideoUrlSubmit()}
              />
            </div>

            <div className="flex gap-2">
              <button
                onClick={handleVideoUrlSubmit}
                className="flex-1 bg-blue-600 hover:bg-blue-700 text-white font-bold py-3 rounded-lg transition-colors"
              >
                Oynat
              </button>
              <button
                onClick={downloadVideo}
                disabled={!videoUrl.trim()}
                className="flex-1 bg-green-600 hover:bg-green-700 disabled:bg-slate-600 text-white font-bold py-3 rounded-lg transition-colors flex items-center justify-center gap-2"
              >
                <Download size={20} />
                İndir
              </button>
            </div>

            <div className="text-xs text-slate-400 text-center mt-4">
              <p>✓ MP4, WebM, Ogg formatları desteklenir</p>
              <p>✓ HTTP/HTTPS URL'leri kullanın</p>
            </div>

            {/* Güvenlik Bilgisi */}
            <button
              onClick={() => setShowSecurityInfo(!showSecurityInfo)}
              className="w-full bg-slate-700 hover:bg-slate-600 text-slate-100 py-2 rounded-lg transition-colors text-xs flex items-center justify-center gap-2"
            >
              <Shield size={16} />
              Güvenlik Bilgisi
            </button>

            {showSecurityInfo && (
              <div className="bg-slate-800 border border-slate-700 rounded-lg p-4 text-xs text-slate-300 space-y-2">
                <p className="font-bold text-green-400 mb-2">🔐 Veri Güvenliği Garantisi:</p>
                <p>✓ Tüm veriler <strong>AES-256</strong> ile şifrelenir</p>
                <p>✓ Hiçbir veri <strong>internete gönderilmez</strong></p>
                <p>✓ <strong>Çevrimdışı çalışır</strong></p>
                <p>✓ <strong>Telemetri yok</strong> - Sizi izlemeyiz</p>
                <p>✓ <strong>Üçüncü parti API yok</strong></p>
                <p>✓ Tüm işlemler <strong>cihazınızda</strong> yapılır</p>
                <p>✓ <strong>Açık kaynak kodlu</strong></p>
              </div>
            )}
          </div>
        </div>
      )}

      {/* Video Player Section */}
      {!showUrlInput && (
        <div className="flex flex-col flex-1">
          {/* Video */}
          <div className={`bg-black flex items-center justify-center ${isFullscreen ? 'w-screen h-screen' : 'w-full aspect-video'}`}>
            <video
              ref={videoRef}
              src={videoUrl}
              onTimeUpdate={handleTimeUpdate}
              onLoadedMetadata={handleLoadedMetadata}
              onEnded={() => setIsPlaying(false)}
              className="w-full h-full object-contain"
              onClick={togglePlayPause}
              style={{
                filter: `brightness(${brightness}%) contrast(${contrast}%) saturate(${saturation}%)`
              }}
            />
          </div>

          {/* Controls */}
          <div className="bg-gradient-to-t from-black via-black to-transparent p-4 space-y-3">
            {/* Progress Bar */}
            <input
              type="range"
              min="0"
              max={duration || 0}
              value={currentTime}
              onChange={handleProgressChange}
              className="w-full h-1 bg-slate-700 rounded-lg appearance-none cursor-pointer accent-blue-600"
              style={{
                background: `linear-gradient(to right, #2563eb 0%, #2563eb ${duration ? (currentTime / duration) * 100 : 0}%, rgb(55, 65, 81) ${duration ? (currentTime / duration) * 100 : 0}%, rgb(55, 65, 81) 100%)`
              }}
            />

            {/* Time Display */}
            <div className="text-xs text-slate-300 px-1">
              {formatTime(currentTime)} / {formatTime(duration)}
            </div>

            {/* İndirme İlerleme Göstergesi */}
            {isDownloading && (
              <div className="space-y-2">
                <div className="flex justify-between items-center text-xs">
                  <span>İndiriliyor...</span>
                  <span>{downloadProgress}%</span>
                </div>
                <div className="w-full bg-slate-700 rounded-full h-2">
                  <div
                    className="bg-green-500 h-2 rounded-full transition-all"
                    style={{ width: `${downloadProgress}%` }}
                  />
                </div>
              </div>
            )}

            {/* Control Buttons */}
            <div className="flex items-center justify-between gap-2">
              {/* Play/Pause & Navigation */}
              <div className="flex gap-2 flex-wrap">
                <button
                  onClick={togglePlayPause}
                  className="p-2 hover:bg-slate-700 rounded-lg transition-colors"
                  title={isPlaying ? "Duraklat" : "Oynat"}
                >
                  {isPlaying ? (
                    <Pause size={24} />
                  ) : (
                    <Play size={24} fill="white" />
                  )}
                </button>

                <button
                  onClick={skipBackward}
                  className="p-2 hover:bg-slate-700 rounded-lg transition-colors"
                  title="10 saniye geri"
                >
                  <SkipBack size={24} />
                </button>

                <button
                  onClick={skipForward}
                  className="p-2 hover:bg-slate-700 rounded-lg transition-colors"
                  title="10 saniye ileri"
                >
                  <SkipForward size={24} />
                </button>

                <button
                  onClick={resetVideo}
                  className="p-2 hover:bg-slate-700 rounded-lg transition-colors"
                  title="Sıfırla"
                >
                  <RotateCcw size={24} />
                </button>

                <button
                  onClick={downloadVideo}
                  disabled={isDownloading}
                  className="p-2 hover:bg-slate-700 disabled:opacity-50 rounded-lg transition-colors"
                  title="Videoyu İndir"
                >
                  <Download size={24} />
                </button>

                <button
                  onClick={() => setShowSettings(!showSettings)}
                  className="p-2 hover:bg-slate-700 rounded-lg transition-colors"
                  title="Ayarlar"
                >
                  <Settings size={24} />
                </button>
              </div>

              {/* Volume & Fullscreen */}
              <div className="flex items-center gap-2">
                <button
                  onClick={toggleMute}
                  className="p-2 hover:bg-slate-700 rounded-lg transition-colors"
                  title={isMuted ? "Sesi aç" : "Sesi kapat"}
                >
                  {isMuted ? (
                    <VolumeX size={24} />
                  ) : (
                    <Volume2 size={24} />
                  )}
                </button>

                <input
                  type="range"
                  min="0"
                  max="1"
                  step="0.1"
                  value={isMuted ? 0 : volume}
                  onChange={handleVolumeChange}
                  className="w-20 h-1 bg-slate-700 rounded-lg appearance-none cursor-pointer accent-blue-600"
                />

                <button
                  onClick={toggleFullscreen}
                  className="p-2 hover:bg-slate-700 rounded-lg transition-colors"
                  title="Tam ekran"
                >
                  <Maximize size={24} />
                </button>
              </div>
            </div>

            {/* Ayarlar Paneli */}
            {showSettings && (
              <div className="bg-slate-900 border border-slate-700 rounded-lg p-4 w-full z-50 shadow-xl">
                <h3 className="text-white font-bold mb-4 flex items-center gap-2">
                  <Zap size={18} /> Görüntü & Ses Ayarları
                </h3>

                {/* Görüntü Ayarları */}
                <div className="space-y-3 mb-4 border-b border-slate-700 pb-4">
                  <div>
                    <label className="text-xs text-slate-300">Parlaklık: {brightness}%</label>
                    <input
                      type="range"
                      min="50"
                      max="150"
                      value={brightness}
                      onChange={(e) => setBrightness(parseInt(e.target.value))}
                      className="w-full h-1 bg-slate-700 rounded-lg accent-blue-500"
                    />
                  </div>
                  <div>
                    <label className="text-xs text-slate-300">Kontrast: {contrast}%</label>
                    <input
                      type="range"
                      min="50"
                      max="150"
                      value={contrast}
                      onChange={(e) => setContrast(parseInt(e.target.value))}
                      className="w-full h-1 bg-slate-700 rounded-lg accent-blue-500"
                    />
                  </div>
                  <div>
                    <label className="text-xs text-slate-300">Doygunluk: {saturation}%</label>
                    <input
                      type="range"
                      min="50"
                      max="150"
                      value={saturation}
                      onChange={(e) => setSaturation(parseInt(e.target.value))}
                      className="w-full h-1 bg-slate-700 rounded-lg accent-blue-500"
                    />
                  </div>
                </div>

                {/* Ses Ayarları */}
                <div className="space-y-3">
                  <div>
                    <label className="text-xs text-slate-300">Bass Boost: {(bass * 100 - 100).toFixed(0)}%</label>
                    <input
                      type="range"
                      min="0.5"
                      max="2"
                      step="0.1"
                      value={bass}
                      onChange={(e) => setBass(parseFloat(e.target.value))}
                      className="w-full h-1 bg-slate-700 rounded-lg accent-green-500"
                    />
                  </div>
                  <div>
                    <label className="text-xs text-slate-300">Treble Boost: {(treble * 100 - 100).toFixed(0)}%</label>
                    <input
                      type="range"
                      min="0.5"
                      max="2"
                      step="0.1"
                      value={treble}
                      onChange={(e) => setTreble(parseFloat(e.target.value))}
                      className="w-full h-1 bg-slate-700 rounded-lg accent-green-500"
                    />
                  </div>
                </div>
              </div>
            )}
          </div>
        </div>
      )}
    </div>
  );
}
